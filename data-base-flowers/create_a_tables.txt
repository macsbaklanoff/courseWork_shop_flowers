DROP SCHEMA public CASCADE;
CREATE SCHEMA public;
CREATE TABLE "Букет" (
	"Идентификатор букета" bigserial,
	"Наименование" text,
	"URL" text,
	"Цена" int,
	"Количество" int,
	PRIMARY KEY ("Идентификатор букета")	
);

CREATE TABLE "Товар"(
	"Идентификатор товара" bigserial,
	"Наименование" text,
	"URL" text,
	"Цена" int,
	"Количество" int,
	PRIMARY KEY ("Идентификатор товара")	
);
CREATE TABLE "Транзакция"(
	"Идентификатор транзакции" bigserial,
	"Идентификатор клиента" int,
	"Идентификатор заказа" int,
	"Способ оплаты" text
	PRIMARY KEY ("Идентификатор транзакции")
	FOREIGN KEY("Идентификатор заказа") REFERENCES "Заказ" ("Идентификатор заказа")	
);
CREATE TABLE "Заказ"(
	"Идентификатор заказа" bigserial,
	"Идентификатор транзакции" int,
	"Идентификатор клиента" int,
	"Заказ" text,
	PRIMARY KEY ("Идентификатор заказа"),
	FOREIGN KEY("Идентификатор клиента") REFERENCES "Клиент" ("Идентификатор клиента")	
);
CREATE TABLE "Клиент"(
	"Идентификатор клиента" bigserial,
	"Фамилия" text,
	"Имя" text,
	"Отчество" text, 
	"Идентификатор заказа" int,
	PRIMARY KEY ("Идентификатор клиента")	
);
